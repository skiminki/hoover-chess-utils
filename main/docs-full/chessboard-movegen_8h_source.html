38template <Color side>
39struct SideSpecificsTempl;
40
41template <>
42struct SideSpecificsTempl<Color::WHITE>
43{
44 static constexpr SquareSet rank1 { SquareSet::row(0U) };
45 static constexpr SquareSet rank2 { SquareSet::row(1U) };
46 static constexpr SquareSet rank3 { SquareSet::row(2U) };
47 static constexpr SquareSet rank4 { SquareSet::row(3U) };
48 static constexpr SquareSet rank5 { SquareSet::row(4U) };
49 static constexpr SquareSet rank6 { SquareSet::row(5U) };
50 static constexpr SquareSet rank7 { SquareSet::row(6U) };
51 static constexpr SquareSet promoRank { SquareSet::row(7U) };
52
53 static constexpr SquareSet pawnAdvance(SquareSet pawns) noexcept
54 {
55 return pawns << 8U;
56 }
57
58 static constexpr SquareSet captureLeft(SquareSet pawns) noexcept
59 {
60 return (pawns & ~SquareSet::column(0U)) << 7U;
61 }
62
63 static constexpr SquareSet captureRight(SquareSet pawns) noexcept
64 {
65 return (pawns & ~SquareSet::column(7U)) << 9U;
66 }
67
68 static constexpr Square captureLeftSq(Square sq) noexcept
69 {
70 assert(columnOf(sq) >= 1U);
71 assert(rowOf(sq) <= 6U);
72 return addToSquareNoOverflowCheck(sq, 7);
73 }
74
75 static constexpr Square captureRightSq(Square sq) noexcept
76 {
77 assert(columnOf(sq) <= 6U);
78 assert(rowOf(sq) <= 6U);
79 return addToSquareNoOverflowCheck(sq, 9);
80 }
81
82 static constexpr Square pawnRetreatSq(Square sq) noexcept
83 {
84 assert(sq >= Square::A3);
85 return addToSquareNoOverflowCheck(sq, -8);
86 }
87
88 static constexpr Square pawnDoubleRetreatSq(Square sq) noexcept
89 {
90 assert(rowOf(sq) == 3U);
91 return addToSquareNoOverflowCheck(sq, -16);
92 }
93};

166SquareSet ChessBoard::blocksAllChecksMask(Square dst) const noexcept
167{
168 const auto numCheckers { m_checkers.popcount() };
169
170 if (numCheckers == 0U) [[likely]]
171 {
172 return SquareSet::all();
173 }
174 else if (numCheckers == 1U)
175 {
176 // do we intercept/capture the check?
177 return
178 (Intercepts::getInterceptSquares(m_kingSq, m_checkers.firstSquare()) & SquareSet::square(dst)).allIfAny();
179 }
180
181 // a single move can never block a double-check
182 return SquareSet::none();
183}

185bool ChessBoard::pinCheck(Square src, Square dst) const noexcept
186{
187 const SquareSet srcBit { SquareSet::square(src) };
188 const SquareSet dstBit { SquareSet::square(dst) };
189
190 return ((m_pinnedPieces & srcBit) == SquareSet::none() ||
191 (Intercepts::getPinRestiction<true>(m_kingSq, src) & dstBit) != SquareSet::none());
192}

210template <typename IteratorType, ChessBoard::MoveGenType type, typename ParamType, Color turn>
211auto ChessBoard::generateMovesForPawnsTempl(
212 IteratorType i,
213 ParamType legalDestinations) const noexcept -> IteratorType
214{
215 using SideSpecifics = SideSpecificsTempl<turn>;
216
217 const SquareSet pawns { m_pawns & m_turnColorMask };
218
219 const SquareSet unpinnedPawns { m_pawns & m_turnColorMask & ~m_pinnedPieces };
220 const SquareSet pinnedPawns { m_pawns & m_turnColorMask & m_pinnedPieces };
221
222 // destination squares for advancing pawns
223 const SquareSet advanceMaskUnpinned { SideSpecifics::pawnAdvance(unpinnedPawns) & ~m_occupancyMask };
224
225 // destination squares for double-advancing pawns
226 const SquareSet doubleAdvanceMaskUnpinned { SideSpecifics::pawnAdvance(advanceMaskUnpinned & SideSpecifics::rank3) & ~m_occupancyMask };
227
228 // pawn advances for unpinned pawns
229 if constexpr (MoveGenIteratorTraits<IteratorType>::storesMoves())
230 {
231 SQUARESET_ENUMERATE(
232 dst,
233 advanceMaskUnpinned & ~SideSpecifics::promoRank & legalDestinations(),
234 {
235 *i = Move { SideSpecifics::pawnRetreatSq(dst), dst, MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
236 ++i;
237 });
238
239 SQUARESET_ENUMERATE(
240 dst,
241 doubleAdvanceMaskUnpinned & legalDestinations(),
242 {
243 *i = Move { SideSpecifics::pawnDoubleRetreatSq(dst), dst, MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
244 ++i;
245 });
246
247 SQUARESET_ENUMERATE(
248 dst,
249 advanceMaskUnpinned & SideSpecifics::promoRank & legalDestinations(),
250 {
251 const Square src { SideSpecifics::pawnRetreatSq(dst) };
252
253 *i = Move { src, dst, pieceToTypeAndPromotion(Piece::QUEEN) };
254 ++i;
255
256 *i = Move { src, dst, pieceToTypeAndPromotion(Piece::ROOK) };
257 ++i;
258
259 *i = Move { src, dst, pieceToTypeAndPromotion(Piece::BISHOP) };
260 ++i;
261
262 *i = Move { src, dst, pieceToTypeAndPromotion(Piece::KNIGHT) };
263 ++i;
264 });
265 }
266 else
267 {
268 i += ((advanceMaskUnpinned | doubleAdvanceMaskUnpinned) &~ SideSpecifics::promoRank & legalDestinations()).
269 popcount();
270
271 i += (advanceMaskUnpinned & SideSpecifics::promoRank & legalDestinations()).popcount() * 4U;
272 }
273
274 const SquareSet oppPieces { (m_occupancyMask ^ m_turnColorMask) };
275 const SquareSet leftCapturingPawns {
276 Attacks::getPawnAttackersMask<turn, false>(oppPieces & legalDestinations()) & pawns };
277 const SquareSet rightCapturingPawns {
278 Attacks::getPawnAttackersMask<turn, true>(oppPieces & legalDestinations()) & pawns };
279
280 if constexpr (MoveGenIteratorTraits<IteratorType>::storesMoves())
281 {
282 // capture left
283 SQUARESET_ENUMERATE(
284 src,
285 leftCapturingPawns & ~SideSpecifics::rank7 & ~m_pinnedPieces,
286 {
287 *i = Move { src, SideSpecifics::captureLeftSq(src), MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
288 ++i;
289 });
290
291 // capture right
292 SQUARESET_ENUMERATE(
293 src,
294 rightCapturingPawns & ~SideSpecifics::rank7 & ~m_pinnedPieces,
295 {
296 *i = Move { src, SideSpecifics::captureRightSq(src), MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
297 ++i;
298 });
299
300 // capture left + promo
301 SQUARESET_ENUMERATE(
302 src,
303 leftCapturingPawns & SideSpecifics::rank7 & ~m_pinnedPieces,
304 {
305 const Square dst { SideSpecifics::captureLeftSq(src) };
306
307 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_QUEEN };
308 ++i;
309
310 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_ROOK };
311 ++i;
312
313 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_BISHOP };
314 ++i;
315
316 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_KNIGHT };
317 ++i;
318 });
319
320 // capture right + promo
321 SQUARESET_ENUMERATE(
322 src,
323 rightCapturingPawns & SideSpecifics::rank7 & ~m_pinnedPieces,
324 {
325 const Square dst { SideSpecifics::captureRightSq(src) };
326
327 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_QUEEN };
328 ++i;
329
330 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_ROOK };
331 ++i;
332
333 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_BISHOP };
334 ++i;
335
336 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_KNIGHT };
337 ++i;
338 });
339 }
340 else
341 {
342 i += (leftCapturingPawns & ~SideSpecifics::rank7 & ~m_pinnedPieces).popcount();
343 i += (rightCapturingPawns & ~SideSpecifics::rank7 & ~m_pinnedPieces).popcount();
344 i += (leftCapturingPawns & SideSpecifics::rank7 & ~m_pinnedPieces).popcount() * 4U;
345 i += (rightCapturingPawns & SideSpecifics::rank7 & ~m_pinnedPieces).popcount() * 4U;
346 }
347
348 // pinned pawns
349 // note: pinned pawns can never resolve a check
350 if constexpr (type == MoveGenType::NO_CHECK)
351 {
352 SQUARESET_ENUMERATE(
353 src,
354 pinnedPawns,
355 {
356 SquareSet pawnBit { SquareSet::square(src) };
357 const SquareSet advanceMask { SideSpecifics::pawnAdvance(pawnBit) & ~m_occupancyMask };
358 const SquareSet doubleAdvanceMask { SideSpecifics::pawnAdvance(advanceMask & SideSpecifics::rank3) & ~m_occupancyMask };
359
360 // advances, non-promo
361 SQUARESET_ENUMERATE(
362 dst,
363 (advanceMask | doubleAdvanceMask) & (~SideSpecifics::promoRank) &
364 Intercepts::getPinRestiction<true>(m_kingSq, src) &
365 legalDestinations(),
366 {
367 *i = Move { src, dst, MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
368 ++i;
369 });
370
371 // note: pinned pawn can never promote without capture
372
373 // left-capture, non-promo
374 SQUARESET_ENUMERATE(
375 dst,
376 oppPieces & SideSpecifics::captureLeft(pawnBit) & (~SideSpecifics::promoRank) &
377 Intercepts::getPinRestiction<true>(m_kingSq, src) & legalDestinations(),
378 {
379 *i = Move { src, dst, MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
380 ++i;
381 });
382
383 // left-capture, promo
384 SQUARESET_ENUMERATE(
385 dst,
386 oppPieces & SideSpecifics::captureLeft(pawnBit) & SideSpecifics::promoRank &
387 Intercepts::getPinRestiction<true>(m_kingSq, src) & legalDestinations(),
388 {
389 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_QUEEN };
390 ++i;
391
392 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_ROOK };
393 ++i;
394
395 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_BISHOP };
396 ++i;
397
398 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_KNIGHT };
399 ++i;
400 });
401
402 // right-capture, non-promo
403 SQUARESET_ENUMERATE(
404 dst,
405 oppPieces & SideSpecifics::captureRight(pawnBit) & (~SideSpecifics::promoRank) &
406 Intercepts::getPinRestiction<true>(m_kingSq, src) & legalDestinations(),
407 {
408 *i = Move { src, dst, MoveTypeAndPromotion::REGULAR_PAWN_MOVE };
409 ++i;
410 });
411
412 // right-capture, promo
413 SQUARESET_ENUMERATE(
414 dst,
415 oppPieces & SideSpecifics::captureRight(pawnBit) & SideSpecifics::promoRank &
416 Intercepts::getPinRestiction<true>(m_kingSq, src) & legalDestinations(),
417 {
418 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_QUEEN };
419 ++i;
420
421 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_ROOK };
422 ++i;
423
424 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_BISHOP };
425 ++i;
426
427 *i = Move { src, dst, MoveTypeAndPromotion::PROMO_KNIGHT };
428 ++i;
429 });
430 });
431 }
432
433 // EP captures, incl. pinned
434 if (m_epSquare <= Square::H8)
435 {
436 const Square epPawn { SideSpecifics::pawnRetreatSq(m_epSquare) };
437 SquareSet checkResolvedOk;
438
439 if constexpr (type == MoveGenType::CHECK)
440 {
441 // single check is resolved if we're capturing the checker
442 checkResolvedOk = (SquareSet::square(epPawn) & m_checkers).allIfAny();
443 }
444 else
445 {
446 static_assert(type == MoveGenType::NO_CHECK);
447 checkResolvedOk = SquareSet::all();
448 }
449
450 SQUARESET_ENUMERATE(
451 src,
452 checkResolvedOk &
453 Attacks::getPawnAttackerMask(m_epSquare, turn) & pawns,
454 {
455 const SquareSet exposedHorizLine {
456 SliderAttacksGeneric::getHorizRookAttackMask(
457 epPawn,
458 m_occupancyMask &~ SquareSet::square(src)) };
459
460 const SquareSet kingBit { m_kings & m_turnColorMask };
461 const SquareSet oppRooks { m_rooks & ~m_turnColorMask };
462
463 if (pinCheck(src, m_epSquare) &&
464 ((kingBit & exposedHorizLine).allIfNone() |
465 (oppRooks & exposedHorizLine).allIfNone()) == SquareSet::all())
466 {
467 *i = Move { src, m_epSquare, MoveTypeAndPromotion::EN_PASSANT };
468 ++i;
469 }
470 });
471 }
472
473 return i;
474}

602template <typename IteratorType, ChessBoard::MoveGenType type, bool shortCastling>
603auto ChessBoard::generateMovesForCastling(
604 IteratorType i) const noexcept -> IteratorType
605{
606 // When in check, castling is not legal. Since the caller is supposed to
607 // classify the position before calling this function, we'll just ensure
608 // here that we're not being called when in check.
609 static_assert(type == MoveGenType::NO_CHECK);
610
611 static_assert(static_cast<std::uint8_t>(Color::WHITE) == 0U);
612 static_assert(static_cast<std::uint8_t>(Color::BLACK) == 8U);
613
614 using CastlingSideSpecifics = CastlingSideSpecificsTempl<shortCastling>;
615
616 const Square sqKing { m_kingSq };
617 Square sqKingTarget;
618 Square sqRookTarget;
619
620 const Color turn { getTurn() };
621 const Square sqRook { getCastlingRook(turn, shortCastling) };
622
623 // do we have rights to castle?
624 if (sqRook == Square::NONE)
625 return i;
626
627 sqKingTarget = makeSquare(CastlingSideSpecifics::kingTargetColumn, (static_cast<std::uint8_t>(turn) / 8U) * 7U);
628 sqRookTarget = makeSquare(CastlingSideSpecifics::rookTargetColumn, (static_cast<std::uint8_t>(turn) / 8U) * 7U);
629
630 const std::uint8_t sqNumKing { static_cast<std::uint8_t>(sqKing) };
631 const std::uint8_t sqNumRook { static_cast<std::uint8_t>(sqRook) };
632 const std::uint8_t sqNumKingTarget { static_cast<std::uint8_t>(sqKingTarget) };
633 const std::uint8_t sqNumRookTarget { static_cast<std::uint8_t>(sqRookTarget) };
634
635 const SquareSet kingPathHalfOpen { BitTricks::rangeHalfOpen(sqNumKing, sqNumKingTarget) };
636 const SquareSet rookPathHalfOpen { BitTricks::rangeHalfOpen(sqNumRook, sqNumRookTarget) };
637
638 const SquareSet requiredEmptySquares {
639 (kingPathHalfOpen | rookPathHalfOpen | SquareSet::square(sqKingTarget) | SquareSet::square(sqRookTarget)) &
640 ~(SquareSet::square(sqKing) | SquareSet::square(sqRook)) };
641
642 // All squares between king and rook are empty? Also, check that the
643 // castling rook is not pinned.
644 if (((requiredEmptySquares & m_occupancyMask) |
645 (SquareSet::square(sqRook) & m_pinnedPieces)) != SquareSet::none())
646 return i;
647
648 // Note: The king's initial square does not need to be checked, since the king
649 // cannot be in check when we enter this function.
650 SquareSet sqMask { (kingPathHalfOpen | SquareSet::square(sqKingTarget)) &~ SquareSet::square(sqKing) };
651
652 SQUARESET_ENUMERATE(
653 sq, sqMask,
654 if (determineAttackers(
655 m_occupancyMask,
656 m_turnColorMask,
657 m_pawns,
658 m_knights,
659 m_bishops,
660 m_rooks,
661 m_kings,
662 sq,
663 turn) != SquareSet::none())
664 return i);
665
666 if constexpr (shortCastling)
667 *i = Move { sqKing, sqRook, MoveTypeAndPromotion::CASTLING_SHORT };
668 else
669 *i = Move { sqKing, sqRook, MoveTypeAndPromotion::CASTLING_LONG };
670 ++i;
671
672 return i;
673}

675template <typename IteratorType, ChessBoard::MoveGenType type, typename ParamType>
676IteratorType ChessBoard::generateMovesTempl(
677 IteratorType i,
678 ParamType legalDestinations) const noexcept
679{
680 // if we're in check, we'll try the king moves first
681 if constexpr (type != MoveGenType::NO_CHECK)
682 {
683 i = generateMovesForKing<IteratorType>(i, m_kingSq);
684 if constexpr (MoveGenIteratorTraits<IteratorType>::canCompleteEarly)
685 if (i.hasLegalMoves())
686 return i;
687 }
688
689 // in double-check, king moves are the only viable moves, so return here
690 if constexpr (type == MoveGenType::DOUBLE_CHECK)
691 return i;
692 else
693 {
694
695 i = generateMovesForPawns<IteratorType, type, ParamType>(i, legalDestinations);
696
697 if constexpr (MoveGenIteratorTraits<IteratorType>::canCompleteEarly)
698 if (i.hasLegalMoves())
699 return i;
700

786 return generateMovesForCastling<IteratorType, type, true>(i);
787 }
788 else
789 return i;

793template <typename IteratorType>
794IteratorType ChessBoard::generateMovesIteratorTempl(
795 IteratorType i) const noexcept
796{
797 const std::uint8_t numCheckers { m_checkers.popcount() };
798 if (numCheckers == 0U) [[likely]]
799 {
800 // all destinations are ok as long as the move is ok
801 i = generateMovesTempl<IteratorType, MoveGenType::NO_CHECK, AllLegalDestinationType>(
802 i, AllLegalDestinationType { });
803 }
804 else [[unlikely]]
805 {
806 if (numCheckers == 1U) [[likely]]
807 {
808 // must capture the checker or block the check
809 i = generateMovesTempl<IteratorType, MoveGenType::CHECK, ParametrizedLegalDestinationType>(
810 i,
811 ParametrizedLegalDestinationType {
812 Intercepts::getInterceptSquares(m_kingSq, m_checkers.firstSquare()) });
813 }
814 else [[unlikely]]
815 {
816 // king moves only
817 i = generateMovesTempl<IteratorType, MoveGenType::DOUBLE_CHECK, NoLegalDestinationType>(
818 i, NoLegalDestinationType { });
819 }
820 }
821
822 return i;
823}
