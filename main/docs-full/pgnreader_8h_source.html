236private:

239 static constexpr std::uint32_t actionBit(PgnReaderActionClass action) noexcept
240 {
241 const unsigned int shift { static_cast<unsigned int>(action) };
242 if (shift < 32U) [[likely]]
243 {
244 return UINT32_C(1) << shift;
245 }
246 else [[unlikely]]
247 {
248 return 0U;
249 }
250 }
251
252 constexpr void enablePack() noexcept
253 {
254 }
255
256 template <typename... Args>
257 constexpr void enablePack(PgnReaderActionClass action, Args... rest) noexcept
258 {
259 set(action, true);
260 enablePack(rest...);
261 }
262
263public:
268 template <typename... Args>
269 constexpr PgnReaderActionFilter(Args... enabled) noexcept
270 {
271 enablePack(enabled...);
272 }
273
278 constexpr inline void set(PgnReaderActionClass action, bool enable) noexcept
279 {
280 const std::uint32_t mask { actionBit(action) };
281
282 if (enable)
283 m_filterBits |= mask;
284 else
285 m_filterBits &= ~mask;
286 }
287
292 constexpr inline bool isEnabled(PgnReaderActionClass action) noexcept
293 {
294 return (m_filterBits & actionBit(action)) != 0U;
295 }
296
301 constexpr inline std::uint32_t getBitMask() noexcept
302 {
303 return m_filterBits;
304 }
305};
