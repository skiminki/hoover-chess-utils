142 template <typename RetType>
143 RetType asciiToUnsigned(const char *str, const char *end, const char *tokenType)
144 {
145 RetType ret { };
146 RetType prev { };
147
148 while (str != end)
149 {
150 char c = *str;
151 ret *= 10U;
152 ret += static_cast<RetType>(c - '0');
153
154 if (ret < prev)
155 throw PgnError(
156 PgnErrorCode::BAD_CHARACTER,
157 std::format("Integer overflow when tokenizing {}", tokenType));
158
159 prev = ret;
160 ++str;
161 }
162
163 return ret;
164 }
