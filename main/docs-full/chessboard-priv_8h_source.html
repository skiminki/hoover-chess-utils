123template <typename IteratorType>
124struct MoveGenIteratorTraits
125{
126 using Iterator = IteratorType;
127
128 // Returns true when further move generation can still add moves. This is
129 // mainly intended to allow movegen with LegalMoveDetectorIterator to exit
130 // early after a legal move has been found.
131 static constexpr bool moveGenCompleted(Iterator) noexcept;
132
133 // Whether this iterator stores moves (ChessBoard::MoveList::iterator) or
134 // whether it's just interested on whether there is a move
135 // (LegalMoveCounterIterator, LegalMoveDetectorIterator). This is used to
136 // skip creating moves is addMoveIfLegal* functions if the actual moves
137 // don't matter.
138 static constexpr bool storesMoves() noexcept;
139};
