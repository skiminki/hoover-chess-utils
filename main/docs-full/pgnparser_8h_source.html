21// Used PGN parsing rules are as follows:
22//
23// PGN = (COMMENT | GAME)* <end_of_file>
24//
25// GAME = TAGPAIRS MOVETEXT
26//
27// TAGPAIRS = (COMMENT | TAGPAIR)*
28// TAGPAIR = <tag_start> <tag_key> <tag_value> <tag_end>
29//
30// MOVETEXT = LINE <result>
31//
32// LINE = COMMENT? (MOVE_ITEM COMMENT? VARIATION*)*
33//
34// VARIATION = <variation_start> LINE <variation_end>
35//
36// MOVE_ITEM = (<movenum_white> | <movenum_black>)? MOVE <nag>*
37//
38// ; note: <move_*> terminals are generally in form:
39// ; - piece
40// ; - source square mask encoding the possible source squares
41// ; - capture
42// ; - destination square
43// ; - promotion piece (Piece::NONE if not a pawn promotion)
44// ; - check/mate mark (ignored)
45// MOVE = <move>
46// | <move_short_castle>
47// | <move_long_castle>
48//
49// COMMENT = <comment_start> (<comment_text> | <comment_newline>)* <comment_end>
50// | <comment_text>
51//
52// See pgnscannertokens.h for the used tokens (terminals). Tokens are here in
53// lowercase in brackets, e.g., <end_of_file>.
54//
55// The rules are derived from the informal description in
56// https://www.saremba.de/chessgml/standards/pgn/pgn-complete.htm .

215 static std::string_view copyToken(const char *src, char *dst, std::size_t maxLen)
216 {
217 std::size_t i;
218 for (i = 0; i < maxLen; ++i)
219 {
220 const char c { src[i] };
221 if (c == '\0')
222 break;
223
224 dst[i] = c;
225 }
226 return std::string_view(dst, i);
227 }

229 void parseTagPair()
230 {
231 strBuilder.clear();
232 strBuilder2.clear();
233 PgnScannerToken token { scanner.nextToken() };
234 if (token != PgnScannerToken::TAG_KEY)
235 throw PgnError(
236 PgnErrorCode::BAD_PGN_TAG,
237 std::format("Expected token TAG_KEY but got: {} ({})",
238 static_cast<std::uint8_t>(token),
239 PgnScanner::scannerTokenToString(token)));
240
241 strBuilder.appendString(scanner.YYText(), scanner.YYLeng());
242
243 token = scanner.nextToken();
244 if (token != PgnScannerToken::TAG_VALUE)
245 throw PgnError(
246 PgnErrorCode::BAD_PGN_TAG,
247 std::format("Expected token TAG_VALUE but got: {} ({})",
248 static_cast<std::uint8_t>(token),
249 PgnScanner::scannerTokenToString(token)));
250
251 // TAG value string has format "...", so we'll crop the first and the
252 // last chars away. In addition, we need to parse escapes
253 bool escape = false;
254 for (const char c : std::string_view { scanner.YYText() + 1, static_cast<std::size_t>(scanner.YYLeng() - 2) })
255 {
256 if (escape)
257 {
258 strBuilder2.pushBack(c);
259 escape = false;
260 }
261 else if (c != '\\')
262 {
263 strBuilder2.pushBack(c);
264 }
265 else
266 escape = true;
267 }
268
269 actionHandler.pgnTag(
270 strBuilder.getStringView(),
271 strBuilder2.getStringView());
272
273 token = scanner.nextToken();
274 if (token != PgnScannerToken::TAG_END)
275 throw PgnError(
276 PgnErrorCode::BAD_PGN_TAG,
277 std::format("Expected token TAG_END but got: {} ({})",
278 static_cast<std::uint8_t>(token),
279 PgnScanner::scannerTokenToString(token)));
280 }

282 PgnScannerToken parseLine(PgnScannerToken token)
283 {
284 bool parsed { };
285
286 token = tryParseComment(token);
287
288 while (true)
289 {
290 std::tie(token, parsed) = tryParseMoveItem(token);
291 if (!parsed)
292 return token;
293
294 if (parsed)
295 {
296 token = tryParseComment(token);
297
298 while (token == PgnScannerToken::VARIATION_START)
299 {
300 token = parseVariation();
301 }
302 }
303 }
304 }
305
306 PgnScannerToken parseVariation()
307 {
308 actionHandler.variationStart();
309
310 PgnScannerToken token { scanner.nextToken() };
311 token = parseLine(token);
312
313 if (token != PgnScannerToken::VARIATION_END)
314 {
315 throw PgnError(
316 PgnErrorCode::UNEXPECTED_TOKEN,
317 std::format("Expected token VARIATION_END but got: {} ({})",
318 static_cast<std::uint8_t>(token),
319 PgnScanner::scannerTokenToString(token)));
320 }
321
322 actionHandler.variationEnd();
323
324 return scanner.nextToken();
325 }

397 std::tuple<PgnScannerToken, bool> tryParseMoveItem(PgnScannerToken token)
398 {
399 bool gotMoveNum { };
400
401 if (token == PgnScannerToken::MOVENUM)
402 {
403 handleMoveNum(scanner.getTokenInfo().moveNum);
404 token = scanner.nextToken();
405 gotMoveNum = true;
406 }
407
408 switch (token)
409 {
410 case PgnScannerToken::MOVE_PAWN:
411 handleMovePawn(scanner.getTokenInfo().pawnMove);
412 break;
413
414 case PgnScannerToken::MOVE_PAWN_CAPTURE:
415 handleMovePawnCapture(scanner.getTokenInfo().pawnMove);
416 break;
417
418 case PgnScannerToken::MOVE_PAWN_PROMO:
419 handleMovePawnPromo(scanner.getTokenInfo().pawnMove);
420 break;
421
422 case PgnScannerToken::MOVE_PAWN_PROMO_CAPTURE:
423 handleMovePawnPromoCapture(scanner.getTokenInfo().pawnMove);
424 break;
425
426 case PgnScannerToken::MOVE_PIECE:
427 switch (scanner.getTokenInfo().pieceMove.piece)
428 {
429 case Piece::KNIGHT:
430 handleMoveKnight(scanner.getTokenInfo().pieceMove);
431 break;
432
433 case Piece::BISHOP:
434 handleMoveBishop(scanner.getTokenInfo().pieceMove);
435 break;
436
437 case Piece::ROOK:
438 handleMoveRook(scanner.getTokenInfo().pieceMove);
439 break;
440
441 case Piece::QUEEN:
442 handleMoveQueen(scanner.getTokenInfo().pieceMove);
443 break;
444
445 default: // KING
446 handleMoveKing(scanner.getTokenInfo().pieceMove);
447 break;
448 }
449 break;
450
451 case PgnScannerToken::MOVE_SHORT_CASTLE:
452 handleMoveShortCastle();
453 break;
454
455 case PgnScannerToken::MOVE_LONG_CASTLE:
456 handleMoveLongCastle();
457 break;
458
459 default:
460 if (gotMoveNum)
461 {
462 throw PgnError(
463 PgnErrorCode::UNEXPECTED_TOKEN,
464 std::format("Expected token MOVE | MOVE_SHORT_CASTLE | MOVE_LONG_CASTLE but got: {} ({})",
465 static_cast<std::uint8_t>(token),
466 PgnScanner::scannerTokenToString(token)));
467 }
468 else
469 {
470 return std::make_tuple(token, false);
471 }
472 }
473
474 token = scanner.nextToken();
475
476 while (token == PgnScannerToken::NAG)
477 {
478 actionHandler.nag(scanner.getTokenInfo().nag.nag);
479 token = scanner.nextToken();
480 }
481
482 return std::make_tuple(token, true);
483 }

485 PgnScannerToken tryParseComment(PgnScannerToken token)
486 {
487 if (token == PgnScannerToken::COMMENT_START)
488 {
489 parseCommentBlock();
490 return scanner.nextToken();
491 }
492 else if (token == PgnScannerToken::COMMENT_TEXT)
493 {
494 parseSingleLineComment();
495 return scanner.nextToken();
496 }
497
498 return token;
499 }
500
501 void parseCommentBlock()
502 {
503 strBuilder.clear();
504
505 std::size_t pendingNewlines { };
506
507 while (true)
508 {
509 const PgnScannerToken token { scanner.nextToken() };
510 switch (token)
511 {
512 case PgnScannerToken::COMMENT_TEXT:
513 if (!strBuilder.isEmpty())
514 {
515 while (pendingNewlines > 0U)
516 {
517 strBuilder.pushBack('\n');
518 --pendingNewlines;
519 }
520 }
521 pendingNewlines = 0U;
522 strBuilder.appendString(scanner.YYText(), scanner.YYLeng());
523 break;
524 case PgnScannerToken::COMMENT_NEWLINE:
525 ++pendingNewlines;
526 break;
527 case PgnScannerToken::COMMENT_END:
528 actionHandler.comment(strBuilder.getStringView());
529 return;
530
531 default:
532 throw PgnError(
533 PgnErrorCode::UNEXPECTED_TOKEN,
534 std::format("Expected token COMMENT_TEXT | COMMENT_NEWLINE | COMMENT_END but got: {} ({})",
535 static_cast<std::uint8_t>(token),
536 PgnScanner::scannerTokenToString(token)));
537 }
538 }
539 }

541 void parseSingleLineComment()
542 {
543 const char *commentStart { scanner.YYText() };
544 const char *commentEnd { commentStart + static_cast<std::size_t>(scanner.YYLeng()) };
545
546 ++commentStart; // eat the ';'
547
548 // eat the white spaces from the beginning
549 while (commentStart != commentEnd)
550 {
551 switch (*commentStart)
552 {
553 case ' ':
554 case '\t':
555 case '\v':
556 ++commentStart;
557 continue;
558
559 default:
560 break;
561 }
562 break;
563 }
564
565 // eat the white spaces from the end
566 while (commentStart != commentEnd)
567 {
568 switch (*(commentEnd - 1U))
569 {
570 case ' ':
571 case '\t':
572 case '\v':
573 --commentEnd;
574 continue;
575
576 default:
577 break;
578 }
579 break;
580 }
581
582 if (commentStart != commentEnd)
583 {
584 actionHandler.comment(std::string_view(commentStart, commentEnd));
585 }
586 }

594
595 void parse()
596 {
597 try
598 {
599 // every iteration parses a game
600 while (true)
601 {
602 PgnScannerToken token { scanner.nextToken() };
603
604 // PGN = (COMMENT | GAME)* <end_of_file>
605 while (true)
606 {
607 if (token == PgnScannerToken::END_OF_FILE)
608 {
609 actionHandler.endOfPGN();
610 return;
611 }
612 else if (token == PgnScannerToken::COMMENT_START)
613 parseCommentBlock();
614 else if (token == PgnScannerToken::COMMENT_TEXT)
615 parseSingleLineComment();
616 else
617 break;
618
619 token = scanner.nextToken();
620 }
621
622 // GAME = TAGPAIRS MOVETEXT
623 actionHandler.gameStart();
624
625 // TAGPAIRS = (COMMENT | TAGPAIR)*
626 while (true)
627 {
628 if (token == PgnScannerToken::TAG_START)
629 parseTagPair();
630 else if (token == PgnScannerToken::COMMENT_START)
631 parseCommentBlock();
632 else if (token == PgnScannerToken::COMMENT_TEXT)
633 parseSingleLineComment();
634 else
635 break;
636
637 token = scanner.nextToken();
638 }
639
640 // MOVETEXT
641 actionHandler.moveTextSection();
642
643 token = parseLine(token);
644
645 if (token != PgnScannerToken::RESULT)
646 {
647 throw PgnError(
648 PgnErrorCode::UNEXPECTED_TOKEN,
649 std::format("Expected token RESULT but got: {} ({})",
650 static_cast<std::uint8_t>(token),
651 PgnScanner::scannerTokenToString(token)));
652 }
653
654 actionHandler.gameTerminated(scanner.getTokenInfo().result.result);
655 }
656 }
657 catch (const PgnError &ex)
658 {
659 // add position info in the exception
660 throw PgnError(scanner, ex);
661 }
662 }
663};
