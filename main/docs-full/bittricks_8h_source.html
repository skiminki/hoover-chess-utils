82 static constexpr inline std::uint64_t isolateHighestSetBit(std::uint64_t mask) noexcept
83 {
84 // Note: we need to 'and' to ensure that the isolated bit actually exists (when n == 0)

119 static inline std::uint64_t parallelExtract(std::uint64_t data, std::uint64_t mask) noexcept
120 {
121#if (HAVE_BMI2)
122 return _pext_u64(data, mask);
123#else
124 std::uint64_t ret { };
125 std::uint64_t count { 1U };
126
127 while (mask)
128 {
129 const std::uint64_t lowestSetBit { BitTricks::isolateLowestSetBit(mask) };
130 ret |= (data & lowestSetBit) ? count : 0U;
131 mask &= ~lowestSetBit;
132 count *= 2U;
133 }
134
135 return ret;
136#endif
137 }

162 static inline std::uint64_t parallelDeposit(std::uint64_t data, std::uint64_t mask) noexcept
163 {
164#if (HAVE_BMI2)
165 return _pdep_u64(data, mask);
166#else
167 std::uint64_t ret { };
168
169 while (mask)
170 {
171 std::uint64_t maskBit { BitTricks::isolateLowestSetBit(static_cast<std::uint64_t>(mask)) };
172
173 ret |= maskBit * (data & 1U);
174 data >>= 1U;
175
176 mask ^= maskBit;
177 }
178
179 return ret;
180#endif
181 }
