224constexpr inline Square makeSquare(std::uint8_t col, std::uint8_t row) noexcept
225{
226 assert(col <= 7U);
227 assert(row <= 7U);
228
229 [[assume(row <= 7U)]];
230 [[assume(col <= 7U)]];
231
232 return static_cast<Square>((row * 8U) + col);
233}

239constexpr inline std::uint8_t columnOf(Square sq) noexcept
240{
241 assert(sq <= Square::H8);
242 [[assume(sq <= Square::H8)]];
243
244 return static_cast<std::uint8_t>(sq) & 7U;
245}

252{

326constexpr inline Color colorOf(PieceAndColor pc) noexcept
327{
328 assert((pc >= PieceAndColor::WHITE_NONE && pc <= PieceAndColor::WHITE_KING) ||
329 (pc >= PieceAndColor::BLACK_NONE && pc <= PieceAndColor::BLACK_KING));
330
331 return Color { static_cast<std::uint8_t>(static_cast<std::uint8_t>(pc) & 0x8U) };
332}

348constexpr inline bool isValidValue(PieceAndColor pc) noexcept

357constexpr inline Piece pieceOf(PieceAndColor pc) noexcept
358{
359 assert(isValidValue(pc));
360 [[assume(pc <= PieceAndColor::BLACK_KING)]];
361
362 return Piece { static_cast<std::uint8_t>(static_cast<std::uint8_t>(pc) & 0x7U) };
363}

370constexpr inline PieceAndColor makePieceAndColor(Piece p, Color c) noexcept
371{
372 assert(isValidValue(p));
373 assert(isValidValue(c));
374
375 return PieceAndColor {
376 static_cast<std::uint8_t>(
377 static_cast<std::uint8_t>(p) | static_cast<std::uint8_t>(c)) };
378}
