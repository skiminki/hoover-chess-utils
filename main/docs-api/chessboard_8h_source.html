268 constexpr Move(Square src, Square dst, MoveTypeAndPromotion typeAndPromo) noexcept :
269 m_encoded(
270 (static_cast<std::uint16_t>(src)) |
271 (static_cast<std::uint16_t>(typeAndPromo) << 6) |
272 (static_cast<std::uint16_t>(dst) << 10U)
273 )
274 {
275 assert(isValidSquare(src));
276 assert(isValidSquare(dst));
277 assert((typeAndPromo <= MoveTypeAndPromotion::PROMO_QUEEN) ||
278 (typeAndPromo == MoveTypeAndPromotion::ILLEGAL));

1218 static constexpr inline std::size_t getCastlingRookIndex(Color c, bool shortCastling) noexcept
