561#define SQUARESET_ENUMERATE_INTERNAL(sq, squareSet, tmpMask, ...) \
562 do \
563 { \
564 SquareSet tmpMask { squareSet }; \
565 \
566 while (true) \
567 { \
568 const Square sq { tmpMask.firstSquare() }; \
569 \
570 if (sq > Square::H8) \
571 break; \
572 \
573 tmpMask &= ~SquareSet::square(sq); \
574 \
575 __VA_ARGS__; \
576 } \
577 } \
578 while (false)
