308 template <std::size_t bufSize, typename UintType>
309 static char *genUnsignedToString(char *s, UintType num) noexcept
310 {
311 static_assert(std::is_unsigned_v<UintType>);
312 static_assert(std::is_integral_v<UintType>);
313 static_assert(std::numeric_limits<UintType>::digits10 + 1U == bufSize);
314
315 char *i = s;
316
317 do
318 {
319 *i++ = '0' + (num % 10U);
320 num /= 10U;
321 }
322 while (num > 0U);
323
324 // Workaround for GCC bug: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=106757
325 //
326 // This should be just:
327 //
328 // std::reverse(s, i);
329
330 const std::size_t len = i - s;
331 [[assume(len <= bufSize)]];
332
333 for (std::size_t c { }; c < len / 2U; ++c)
334 std::swap(s[c], s[len - c - 1U]);
335
336 return i;
337 }

346 static MiniString<13U> moveNumToString(std::uint32_t moveNum, Color turn) noexcept
347 {
348 // max length: "4294967295..."
349 // 1234567890123 = 13 chars
350 MiniString<13U> ret { MiniString_Uninitialized() };
351
352 char *i = ret.data();
353
354 i = genUnsignedToString<10U>(i, moveNum);
355
356 *i++ = '.';
357 if (turn == Color::BLACK)
358 {
359 *i++ = '.';
360 *i++ = '.';
361 }
362
363 ret.setLength(i - ret.data());
364 return ret;
365 }

372 static constexpr char promoPieceChar(Piece promo) noexcept
373 {
374 static_assert(Piece::KNIGHT == Piece { 2U });
375 static_assert(Piece::BISHOP == Piece { 3U });
376 static_assert(Piece::ROOK == Piece { 4U });
377 static_assert(Piece::QUEEN == Piece { 5U });
378
379 std::uint32_t pieceCharactersWord {
380 (static_cast<std::uint32_t>('R') << 0U) |
381 (static_cast<std::uint32_t>('Q') << 8U) |
382 (static_cast<std::uint32_t>('N') << 16U) |
383 (static_cast<std::uint32_t>('B') << 24U) };
384
385 unsigned int shift { (static_cast<std::uint8_t>(promo) & 3U) * 8U };
386 return static_cast<char>((pieceCharactersWord >> shift) & 0xFFU);
387 }

428 static constexpr MiniString<1U> pieceToSanStr(Piece p) noexcept
429 {
430 static_assert(Piece::NONE == Piece { 0U });
431 static_assert(Piece::PAWN == Piece { 1U });
432 static_assert(Piece::KNIGHT == Piece { 2U });
433 static_assert(Piece::BISHOP == Piece { 3U });
434 static_assert(Piece::ROOK == Piece { 4U });
435 static_assert(Piece::QUEEN == Piece { 5U });
436 static_assert(Piece::KING == Piece { 6U });
437
438 constexpr std::uint64_t pieceCharactersWord {
439 (static_cast<std::uint64_t>('?') << 0U) |
440 (static_cast<std::uint64_t>('N') << 16U) |
441 (static_cast<std::uint64_t>('B') << 24U) |
442 (static_cast<std::uint64_t>('R') << 32U) |
443 (static_cast<std::uint64_t>('Q') << 40U) |
444 (static_cast<std::uint64_t>('K') << 48U) };
445
446 if (p > Piece::KING)
447 p = Piece::NONE;
448
449 const unsigned int index { static_cast<unsigned int>(p) };
450
451 char pieceCharacter { static_cast<char>((pieceCharactersWord >> (index * 8U)) & 0xFFU) };
452
453 MiniString<1U> ret { MiniString_Uninitialized() };
454 ret[0U] = pieceCharacter;
455 ret.setLength(pieceCharacter != '\0' ? 1U : 0U);
456
457 return ret;
458 }
